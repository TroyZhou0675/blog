（以f4系列芯片为例）
```
void SystemInit(void)
{
    /*---------------------------------------------*/
    /* ✅ 1. 启用 FPU（浮点处理单元）              */
    /*---------------------------------------------*/
#if (__FPU_PRESENT == 1) && (__FPU_USED == 1)
    SCB->CPACR |= ((3UL << (10 * 2)) | (3UL << (11 * 2)));  // CP10, CP11 Full Access
#endif

    /*---------------------------------------------*/
    /* ✅ 2. 复位 RCC 时钟配置为默认值              */
    /*---------------------------------------------*/

    RCC->CR |= 0x00000001;         // 打开 HSI（内部时钟）
    RCC->CFGR = 0x00000000;        // 清除时钟配置
    RCC->CR &= 0xFEF6FFFF;         // 关闭 HSE、PLL、CSS
    RCC->PLLCFGR = 0x24003010;     // 恢复 PLL 默认配置
    RCC->CR &= 0xFFFBFFFF;         // 清除 HSEBYP（关闭旁路）
    RCC->CIR = 0x00000000;         // 禁用所有 RCC 中断

    /*---------------------------------------------*/
    /* ✅ 3. 初始化外部存储器控制器（如使用 SDRAM） */
    /*---------------------------------------------*/
#if defined (DATA_IN_ExtSRAM) || defined (DATA_IN_ExtSDRAM)
    SystemInit_ExtMemCtl();  // 外部内存控制器初始化（可选）
#endif

    /*---------------------------------------------*/
    /* ✅ 4. 设置系统时钟 SYSCLK                    */
    /*---------------------------------------------*/
    SetSysClock();  // 设置 PLL、分频器、FLASH 延时等 → 决定 SystemCoreClock

    /*---------------------------------------------*/
    /* ✅ 5. 设置中断向量表的偏移地址               */
    /*---------------------------------------------*/
#ifdef VECT_TAB_SRAM
    SCB->VTOR = SRAM_BASE | VECT_TAB_OFFSET;   // 中断向量表重定位到 SRAM
#else
    SCB->VTOR = FLASH_BASE | VECT_TAB_OFFSET;  // 默认使用 Flash 中的向量表
#endif
}

```
1. 启用浮点计算单元FPU
2. 初始化RCC和和内部存储控制器
3. 设置时钟（最重要）
4. 设置中断向量表偏移地址
