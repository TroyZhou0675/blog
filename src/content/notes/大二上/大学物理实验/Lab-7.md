$\overline{D} = \frac{\sum_{n=1}^{6} D_n }{6}= 121.19cm$
$\overline{d_1} = 3.414mm$
$\overline{d_2} = 1.498mm$
$d = \sqrt{\overline{d_1}\cdot\overline{d_2}} = 2.261mm$
$u_A(D)=\sqrt{\frac{1}{n-1}\sum_{i=1}^{n}\bigl(D_i-\bar{D}\bigr)^2} =0.16cm$
$u_B(D)=\frac{\Delta_{\text{仪}}}{\sqrt{3}}=\frac{0.2\ \text{mm}}{\sqrt{3}}=0.115mm$
$u_D=\sqrt{u_A(D)^2+u_B(D)^2}=0.16cm$
$D=\bar{D}\pm u_D=121.19\pm0.16cm$
$u_A(d_1)=s_{d_1}=\sqrt{\frac{1}{n-1}\sum_{i=1}^{n}(d_{1i}-\bar{d}_1)^2} =0.047mm$
$u_A(d_2) = s_{d_2}=\sqrt{\frac{1}{n-1}\sum_{i=1}^{n}(d_{2i}-\bar{d}_2)^2}=0.008mm$
$u_B=\frac{\Delta_{\rm 仪}}{\sqrt{3}}=\frac{0.004\ \mathrm{mm}}{\sqrt{3}}=0.0013mm$
$u_{d_1}=\sqrt{\,u_A(d_1)^2+u_B^2\,}=0.047mm$
$u_{d_2}=\sqrt{\,u_A(d_2)^2+u_B^2\,}=0.008mm$
$d_1=\bar{d}_1\pm u_{d_1}=3.414\pm0.047mm$
$d_2=\bar{d}_2\pm u_{d_2}=1.498\pm0.008mm$

$\overline{y} =10\overline{\Delta x}=3.520mm$
$\overline{\lambda}=\frac{\sqrt{\overline{d_1}}\cdot\sqrt{\overline{d_2}}}{\overline{D}}*\frac{\overline{y}}{10} = 656.8nm$
$u_A(y) = s_y = \sqrt{\frac{1}{n-1}\sum_{i=1}^6 (y_i-\overline{y})^2} =0.034mm$
$u_B=\frac{0.004\ \mathrm{mm}}{\sqrt{3}}$
$u_y=\sqrt{\,u_A(y)^2+u_B^2\,}=0.034mm$
$y=\bar{y}\pm u_y = 3.520\pm0.034mm$
$\frac{u_\lambda}{\lambda} = \sqrt{\left(\frac{u_{d_1}}{2d_1}\right)^2 +\left(\frac{u_{d_2}}{2d_2}\right)^2 +\left(\frac{u_y}{y}\right)^2 +\left(\frac{u_D}{D}\right)^2} = 0.0122$
$u_{\lambda}=0.0122\times\lambda = 8.0nm$
$\lambda=\overline\lambda\pm u_{\lambda}=656.8\pm8.0nm$
