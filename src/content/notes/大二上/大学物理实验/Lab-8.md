$R=\frac{d_m^2-d_n^2}{4(m-n)\lambda}$
 $\overline{R}=\frac1 6 \sum_{i=1}^{6}R_i =1.024m$
 
 $S(R)=\sqrt{\frac{1}{6-1}\sum_{i=1}^{6}(R_i-\bar{R})^2} =0.0059m$
 $u(R)=\frac{S(R)}{\sqrt6}=0.002m$
 $R = \bar{R}\pm u(R)=1.024\pm0.002m$

$L_1 = 45.815mm$
$L_2 = 6.679mm$
$L = L_1-L_2=39.136mm$

$e=\frac{5\lambda L}{s_{x+10}-s_x}$
$\bar{e}=\frac1 {10}\sum_{i=1}^10=0.0405mm$

$S(\Delta S)=\sqrt{\frac{1}{10-1}\sum_{i=1}^{10}(\Delta S_i-\bar{\Delta S})^2} =0.0795mm$
$u_A(\Delta S)=\frac{S(\Delta S)}{\sqrt{10}}=0.0251mm$
$u_B(\Delta S)=\frac{0.004mm}{\sqrt3}=0.0023mm$
$u(\Delta S)=\sqrt{u_A(\Delta S)^2+u_B(\Delta S)^2}=0.0252mm$

$u(L) =\frac{0.004mm}{\sqrt3}=0.0023mm$
$$\frac{u(e)}{e}
=
\sqrt{
\left( \frac{u(\lambda)}{\lambda} \right)^2
+
\left( \frac{u(L)}{L} \right)^2
+
\left( \frac{u(\Delta S)}{\Delta S} \right)^2
}$$
$u(e) = 0.0004mm$
$e=\bar{e}\pm u(e)=0.0405\pm0.0004 mm$
